combined_df <- dplyr::bind_rows(lapply(file_list, read.csv))

combined_df$avg_rating <- (combined_df$rating_first_watch + combined_df$rating_second_watch) / 2

unique_movies <- unique(combined_df$title)

aggregated_data <- data.frame()

for (title in unique_movies) {
 
  df_subset <- combined_df[combined_df$title == title, ]
 
  avg_rating <- mean(df_subset$avg_rating)
  total_reviews <- nrow(df_subset)
 
  movie_info <- df_subset[1, c("title", "release_year", "genre")]
 
 
  aggregated_movie <- data.frame(
    title = movie_info$title,
    release_year = movie_info$release_year,
    genre = movie_info$genre,
    AverageRating = avg_rating,
    TotalReviews = total_reviews
  )
 
  aggregated_data <- rbind(aggregated_data, aggregated_movie)
}
